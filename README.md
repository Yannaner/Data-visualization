# Data-visualization
A data  visualization presentation for CS 131

Different Marker Styles:
marker_styles = [
    '.',  # Point
    ',',  # Pixel
    'o',  # Circle
    '^',  # Triangle up
    'v',  # Triangle down
    '<',  # Triangle left
    '>',  # Triangle right
    's',  # Square
    'D',  # Diamond
    '*',  # Star
    'p',  # Pentagon
    'h',  # Hexagon1
    'H',  # Hexagon2
    '+',  # Plus
    'x'   # Cross
]
